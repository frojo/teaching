Coroutines

Coroutines are special functions that execute code over time.

They’re used to code anything that happens over time

  • timers
  • dialog
  • turn order in strategy games
  • Enemy Behavior
  • screenshake and other juicy things

Coroutines look like other functions, except they have an IEnumerator type.

Here’s how you would write a coroutine that prints something after 5 seconds.

using UnityEngine;
using System.Collections;

public class ExampleScript : MonoBehaviour
{

    void Start()
    {
        // - After 0 seconds, prints "Starting 0.0 seconds"
        
        print("Starting " + Time.time + " seconds");
        // Start function WaitAndPrint as a coroutine.
        StartCoroutine(WaitAndPrint());
    }

    private IEnumerator WaitAndPrint()
    {
        yield return new WaitForSeconds(5);

				// - After 5 seconds, prints "Coroutine ended: 2.0 seconds"

        print("Coroutine ended: " + Time.time + " seconds");
    }
}

Screenshake code

using System.Collections;
using System.Collections.Generic;
using UnityEngine;

public class Screenshake : MonoBehaviour
{

    public int length;
    public float intensity;

    void Start()
    {

    }

    public void DoShakeScreen() {
        StartCoroutine(ShakeScreen());
    }

    private IEnumerator ShakeScreen()
    {

        for (int frames = 0; frames <= length; frames++)
        {
            transform.position += new Vector3(Random.Range(-intensity, intensity),
                                              Random.Range(-intensity, intensity),
                                             0);
            yield return null; 
        }

 
    }

}
